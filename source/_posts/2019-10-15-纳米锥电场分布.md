```matlab
%% 将FDTD数据导入到Matlab中绘制电场分布图，对应文件：cone20191014.m
X = x*10^9;  %x是FDTD输出的x坐标
Y = y*10^9;  %x是FDTD输出的y坐标
h = pcolor(X,Y,E2.');  %E2是FDTD输出的电场平方
set(h,'LineStyle','none');      %线型设置为无 
axis equal  %x，y的尺寸等比例
 axis([-40 40 -40 40])  %规定x，y取值范围
set(gca,"FontSize",25,"Fontweight",'bold')  设定当前图层字体大小，粗细
set(gca,'XTick',(-40:20:40))  %设定x方向的刻度标注
set(gca,'YTick',(-40:20:40))  %设定y方向的刻度标注
shading interp  %对曲面或图形对象的颜色着色进行色彩的插值处理，使色彩平滑过渡
xlabel('x (nm)')  %x标题
ylabel('y (nm)')  %y标题
colorbar  %显示colorbar
colormap jet  %colorbar类型选择为jet
```
