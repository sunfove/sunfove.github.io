python\nimport numpy as np\nimport matplotlib.pyplot as plt\nfrom scipy.fft import fft, fftfreq\n\n# 参数设置\nfs = 1000          # 采样率 (Hz)\nT = 1.0            # 信号时长 (s)\nn = int(fs * T)    # 采样点数\nt = np.linspace(0, T, n, endpoint=False)\n\n# 合成信号：10 Hz + 50 Hz 正弦波\nx = 0.7*np.sin(2*np.pi*10*t) + 1.0*np.sin(2*np.pi*50*t)\n\n# 计算 FFT\nX = fft(x)\nfreqs = fftfreq(n, 1/fs)\nhalf_n = n // 2\n# 取单边谱\nmagnitude = 2.0/n * np.abs(X[:half_n])\nplt.figure(figsize=(10,4))\nplt.plot(freqs[:half_n], magnitude)\nplt.title('Signal Spectrum (no window)')\nplt.xlabel('Frequency (Hz)')\nplt.ylabel('Magnitude')\nplt.grid(True)\nplt.show()\n\n# 加汉宁窗\nwindow = np.hanning(n)\nx_windowed = x * window\nX_win = fft(x_windowed)\nmagnitude_win = 2.0/n * np.abs(X_win[:half_n])\nplt.figure(figsize=(10,4))\nplt.plot(freqs[:half_n], magnitude_win)\nplt.title('Signal Spectrum (Hanning window)')\nplt.xlabel('Frequency (Hz)')\nplt.ylabel('Magnitude')\nplt.grid(True)\nplt.show()\n
